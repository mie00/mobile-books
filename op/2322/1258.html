مثلاًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًًً بتعلم السباحِة بمواجهةِ خطرِ الغَرَقِ .
لا قفل إلا سوف يُفْتَحُ ، ولا قيد إلا سوف يُفَكٌّ، ولا بعيد إلا سوف يقربُ ، ولا غائب إلا سوف يصلُ.. ولكن بأجل مسمّى .
? اسْتَعِينُوا بِالصَّبْرِ وَالصَّلاةِ ? فهما وَقودُ الحياةِ ، وزادُ السيرِ ، وباب الأملِ ، ومفتاحُ الفَرَجِ ، ومن لزم الصبرَ ، وحافظ على الصلاةِ ؛ فبشِّرْه بفجرٍ صادقٍ ، وفتحٍ مبينٍ ، ونصرٍ قريبٍ .
